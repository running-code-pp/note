# 头文件
## 防止重复包含
#ifndef <PROJECT>_<PATH>_<FILE>_H_ 
#define <PROJECT>_<PATH>_<FILE>_H_
...
#endif

# 不希望对外暴露的细节
放在details命名空间中，并加上__作为前缀，例如
```cpp
namespace async_simple {
namespace detail {
template <class>
struct __void_t {
    typedef void type;
};
}  // namespace detail
}  // namespace async_simple

```
